
\lingset{exnoformat=X,glhangstyle=none}
\newcount\scnt
% \scntinit to initialize the sentence number counter
\def\scntinit#1 {\scnt=#1}
\newdimen\swidth
% \setswidth establishes a gutter for the sentence number (#1), a following
%    period, a gap between the period and the name (#2), and the name (#3)
\def\setswidth #1 #2 #3 {\setbox0=\hbox{#1.\hskip#2 #3}%
   \swidth=\wd0}
% \speaker uses ExPex's \exdisplay to typeset the sentences
\def\speaker#1{%
   \exdisplay
   \leftskip=\swidth
   \advance\leftskip by \lingtextoffset
   \leavevmode
   \llap{\hbox to \leftskip{\the\scnt.\hfil #1\hskip\lingtextoffset}}%
   \global\advance\scnt by 1
   \ignorespaces
}
\let\endspeaker\xe
