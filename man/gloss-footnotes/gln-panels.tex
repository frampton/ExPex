\def\gln@panel@endgl{%

   \vtop\bgroup
   \hsize=\ssrightwd
   \leftskip=0pt
   \rightskip=\lingssrightskip
   \lingeverypanel
}%
\def\begin@glpanel@right{%
   \ling@usearg
   \ep@setssdims
   \leavevmode

}

   \bgroup
   \ifcase\gl@stylenum
      \let\endgl=\glw@panel@endgl
   \else
      \let\endgl=\gln@panel@endgl
   \fi
   \begin@glpanel@right
}
