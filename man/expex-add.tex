\def\glw@begingl@a{%
   \ifling@allowwidegl
         \hbox\bgroup\vtop\bgroup \advance\hsize by \lingextraglwidth
      \else
         \vtop\bgroup
      \fi
      \ling@usearg
      \@glpreamblefalse
      \ifdim\ling@glwidth=0pt
            \advance\hsize by -\leftskip
            \advance\hsize by -\rightskip
         \else
            \hsize=\ling@glwidth
         \fi
      \ling@everygl
      \leftskip=0pt
      \rightskip=\ling@glrightskip
      \ifcase\ep@glhangstyle
         \or
            \hangindent=\ling@glhangindent
            \hangafter=1
         \or
            \glwcascade@makeshape
            \parshape 9
            \glwcascade@parshapetarget
         \fi
      \let\mainlist=\empty
      \let\aboveskiplist=\empty
      \let\strutlist=\empty
      \let\everylist=\empty
}
\def\glw@endglB{%
   \ifling@allowwidegl \egroup\hskip-\lingextraglwidth \fi
    \egroup\egroup}
